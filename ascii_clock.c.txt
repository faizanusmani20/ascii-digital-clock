#include <stdio.h>
#include <time.h>
#include <unistd.h>

/* ASCII Art Digital Clock
 * -----------------------
 * Colored header, date and time display
 */

#define CYAN   "\033[36m"
#define YELLOW "\033[33m"
#define RESET  "\033[0m"
#define CLEAR  "\033[2J\033[H"

int main() {
    time_t now;
    struct tm *local;
    char buffer[50];

    while (1) {
        time(&now);
        local = localtime(&now);

        printf(CLEAR);

        /* ASCII ART HEADER */
        printf(CYAN);
        printf("  ██████╗██╗      ██████╗  ██████╗██╗  ██╗\n");
        printf(" ██╔════╝██║     ██╔═══██╗██╔════╝██║ ██╔╝\n");
        printf(" ██║     ██║     ██║   ██║██║     █████╔╝ \n");
        printf(" ██║     ██║     ██║   ██║██║     ██╔═██╗ \n");
        printf(" ╚██████╗███████╗╚██████╔╝╚██████╗██║  ██╗\n");
        printf("  ╚═════╝╚══════╝ ╚═════╝  ╚═════╝╚═╝  ╚═╝\n");
        printf(RESET);

        printf("\n");

        /* DATE */
        strftime(buffer, sizeof(buffer), "%A, %d %B %Y", local);
        printf(YELLOW "        %s\n\n" RESET, buffer);

        /* TIME (12-hour format with AM/PM) */
        strftime(buffer, sizeof(buffer), "%I:%M:%S %p", local);
        printf(CYAN "            %s\n" RESET, buffer);

        fflush(stdout);
        sleep(1);
    }

    return 0;
}
